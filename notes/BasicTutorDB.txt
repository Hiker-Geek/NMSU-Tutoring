Create table users
	(aggieID		int 		NOT NULL UNIQUE,
	 FirstName		varchar(32),
	 LastName		varchar(32),
	 Password		varchar(32),
	 Birthdate	   	 date,
	 Email			varchar(32)	NOT NULL,
	 Role			char(1)		NOT NULL,
	 Gender		char(1),
	 Bio		varChar(32),
	 DeptID		int,
	 Subject		varchar(32),
	 primary key (aggieID)
	);

Create table tutors
(aID			int			NOT NULL,
	userTutorID		int			NOT NULL UNIQUE,
	DeptID		int,
	Subject		varchar(32),
	 primary key (userTutorID),
	 foreign key (aID) references users(aggieID)
		on delete cascade
);

Create table students
(aID			int			NOT NULL,
userStudentID		int			NOT NULL UNIQUE,
	 primary key (userStudentID),
	 foreign key (aID) references users(aggieID)
		on delete cascade
);
	
	
	create table schedules
    (scheduleID		int	NOT NULL UNIQUE,
     scheduledTutorID  int,
     meetTime       dateTime NOT NULL,
     endTime        dateTime,
     building		varchar(32),
     roomNumber	    varchar(10),
     notes 		varchar(128),
     primary key (scheduleID)
         foreign key (scheduledTutorID) references tutors(userTutorID)
         on delete cascade
    );

	
	create table booking
	(appointmentID			int 	NOT NULL UNIQUE,
	studentAggieID			int,
	tutorAggieID			int,
	schedulingID			int,
	BookDate				date,
	BookTime				time,
	primary key (appointmentID),
	foreign key (studentAggieID) references users(aggieID)
		on delete cascade,
	foreign key (tutorAggieID) references tutors(aggieID)
		on delete cascade,
	 foreign key (schedulingID) references schedules(scheduleID)
		on delete cascade
	);
