create table users
	(uID				int,
	 FirstName		varchar(32),
	 LastName		varchar(32),
	 Birthdate	    date,
	 Email			varchar(32),
	 DeptID				int,
	 Category		varchar(32),
	 TOS			char(1),
	 Gender			char(1),
	 primary key (uID)
	);
	
	create table schedules
	(sID				int,
	 meetDate		date,
	 meetTime		time,
	 building		varchar(32),
	 roomNumber	    varchar(10),
	 primary key (sID)
	);
	
	create table booking
	(userID	int,
	 scheduleID			int,
	 BookDate		date,
	 BookTime		time,
	 primary key (userID, scheduleID),
	 foreign key (userID) references users(uID)
		on delete cascade,
	 foreign key (scheduleID) references schedules(sID)
		on delete cascade
	);
